SET(KIT MigrateSimmKEModel)
SET(UKIT MIGRATESIMMKEMODEL)

##
## Version stuff
##
SET(OPENSIM_MIGRATESIMMKEMODEL_LIBRARY_NAME osim${KIT})
SET(OPENSIM_MIGRATESIMMKEMODEL_MAJOR_VERSION 0)
SET(OPENSIM_MIGRATESIMMKEMODEL_MINOR_VERSION 7)
SET(OPENSIM_MIGRATESIMMKEMODEL_BUILD_VERSION 1)
SET(OPENSIM_MIGRATESIMMKEMODEL_COPYRIGHT_YEARS "2008")
# comma separated list of authors
SET(OPENSIM_MIGRATESIMMKEMODEL_AUTHORS  "Peter_Loan_and_Ajay_Seth") 


ADD_DEFINITIONS(-DOPENSIM_MIGRATESIMMKEMODEL_LIBRARY_NAME=${OPENSIM_MIGRATESIMMKEMODEL_LIBRARY_NAME}
-DOPENSIM_MIGRATESIMMKEMODEL_MAJOR_VERSION=${OPENSIM_MIGRATESIMMKEMODEL_MAJOR_VERSION}
-DOPENSIM_MIGRATESIMMKEMODEL_MINOR_VERSION=${OPENSIM_MIGRATESIMMKEMODEL_MINOR_VERSION}
-DOPENSIM_MIGRATESIMMKEMODEL_BUILD_VERSION=${OPENSIM_MIGRATESIMMKEMODEL_BUILD_VERSION}
-DOPENSIM_MIGRATESIMMKEMODEL_COPYRIGHT_YEARS="${OPENSIM_MIGRATESIMMKEMODEL_COPYRIGHT_YEARS}"
-DOPENSIM_MIGRATESIMMKEMODEL_AUTHORS="${OPENSIM_MIGRATESIMMKEMODEL_AUTHORS}"
-DOPENSIM_MIGRATESIMMKEMODEL_TYPE="Shared")

FILE(GLOB SOURCE_FILES *.cpp)
FILE(GLOB INCLUDE_FILES *.h)

INCLUDE_DIRECTORIES(${OpenSim_SOURCE_DIR} ${OpenSim_SOURCE_DIR}/Vendors ${XERCES_INCLUDE_DIR})

SET(EXPORT_MACRO OSIM${UKIT}_EXPORTS)

LINK_LIBRARIES( osimSimmKinematicsEngine osimSimbodyEngine )

IF (BUILD_STATIC_OSIM_LIBS)
	ADD_LIBRARY(osim${KIT} STATIC ${SOURCE_FILES} ${INCLUDE_FILES}) 
	ADD_DEFINITIONS(-DSTATIC_OSIM_LIBS)
ELSE(BUILD_STATIC_OSIM_LIBS)
	ADD_LIBRARY(osim${KIT} SHARED ${SOURCE_FILES} ${INCLUDE_FILES}) 
	SET_TARGET_PROPERTIES(osim${KIT} PROPERTIES
		DEFINE_SYMBOL ${EXPORT_MACRO}
		PROJECT_LABEL "Libraries - osim${KIT}"
	)
ENDIF(BUILD_STATIC_OSIM_LIBS)

INSTALL_TARGETS(/sdk/lib/ RUNTIME_DIRECTORY /bin osim${KIT})
